    }
    
    try:

        print(f"ğŸ“¡ Response Status: {response.status_code}")
        
        if response.status_code == 200:
            data = response.json()
            print("âœ… SUCCESS! Data berhasil diambil")
            
            if 'm2m:cin' in data and 'con' in data['m2m:cin']:
                content_str = data['m2m:cin']['con']
                try:
                    # Parse JSON string content
                    content = json.loads(content_str)
                    print(f"ğŸ“Š Data terbaru: {json.dumps(content, indent=2)}")
                    return content
                except json.JSONDecodeError:
                    # Jika bukan JSON, tampilkan sebagai string
                    print(f"ğŸ“Š Data terbaru (raw): {content_str}")
                    return content_str
            else:
                print("ğŸ“­ Struktur data tidak sesuai atau device kosong")
                print(f"ğŸ“‹ Raw response: {json.dumps(data, indent=2)}")
        else:
            print(f"âŒ Status tidak OK: {response.text}")
            
    except Exception as e:
        print(f"âŒ Error: {str(e)}")
    
    return None
